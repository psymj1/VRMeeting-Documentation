def upload_file
        # check presence
        if is_params_present && is_request_valid
                # create a record
                f = FileList.new
                f.file = get_tempfie
                f.Mid = get_mid
                
                # save the record (save the file)
                if   f.save!
                        # convert to images
                        pdf_to_image(f)
                        # remove the original pdf and the record
                        f.remove_file!
                        f.destroy
                        render status: :ok
                else
                        render status: :bad_request
                end
        else
                render status: :bad_request
        end
end
